cmake_minimum_required(VERSION 3.16)
project(Fractoid)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_AUTOUIC ON)
set(CMAKE_CXX_FLAGS ${CMAKE_CXX_FLAGS} -pthread)
set(REQUIRED_LIBS_QUALIFIED Qt5::Core Qt5::Gui Qt5::Widgets)
find_package(OpenCV REQUIRED)
find_package(Qt5 COMPONENTS Core Gui Widgets REQUIRED)
include_directories(${OpenCV_INCLUDE_DIR})
add_executable(fractoid src/main.cpp src/Fractal.cpp src/Fractal.h src/Mandelbrot.cpp src/Mandelbrot.h src/BurningShip.cpp src/BurningShip.h src/Julia.cpp src/Julia.h src/Phoenix.cpp src/Phoenix.h src/Manowar.cpp src/Manowar.h src/Newton.cpp src/Newton.h src/MainWindow.cpp src/MainWindow.h)
target_link_libraries(fractoid ${OpenCV_LIBS} ${REQUIRED_LIBS_QUALIFIED})
